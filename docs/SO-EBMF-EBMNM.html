plotL <- function(A.l, title){

  K <- ncol(A.l)
  n <- nrow(A.l)
  
  data.frame(L=c(A.l),
             factor=rep(1:K, each=n),
             x=rep(1:n, times=K)) %>%
    mutate(factor=as.factor(factor)) -> df.plot
  
  df.plot %>%
    ggplot()+
    geom_line(aes(x=x, y=L, group=factor, col=factor))+
    facet_wrap(~factor, scales='free_y') +ggtitle(title)-> figL
  
  print(figL)
}
